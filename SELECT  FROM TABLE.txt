SAME PRODUCT BROUGHT BY CUSTOMER IN CONSUQUITE MONTH?

WITH PURCHASE AS (
SELECT CUST_ID,
CUST_NAME,
PROD_ID,
PROD_NAME,
DATE_TRUNC('MONTH',PUR_DATE) AS PURCHASE_MONTH,
LAG(DATE_TRUNC('MONTH',PUR_DATE)) OVER(
PARTITION BY CUST_ID,PROD_ID
ORDER BY PUR_DATE
)
AS PREV_MONTH
FROM PURCHASE P
JOIN CUST C ON P.CUST_ID=C.CUST_ID
JOIN PRODUCT P ON P.PROD_ID=P.PROD_ID
)
SELECT DISTINCT CUST_ID,CUST_NAME,PROD_ID,PROD_NAME
FROM PURCHASE_MONTH
WHERE DATEDIFF(MONTH,PREV_MONTH,PURCHSE_MANTH) = 1;

with source as (
select cust_id,
name,
address,
membership_tier,
updated_at
from {{ ref('stg_customers') }}
),
scd updates as (
s.customer_id,
s.name,
s.address,
s.membership_tier,
s.updated_at,
coalesce(d.customer_scd_id,
md5(cast(s.customer_id as string)|| '-' || cast(s.updated_at as string ))) as customer_scd_id,
case
when d.customer_scd_id is null then true -- new record
when s.updated_at> d.update_at then true -- change record
else false
end as is_new_record
from source s
left join {{this}} d
on s.customer_id = d.customer_id
and d.is_active = true -- join with active records only
)
select customer_scd_id,
customer_id,
name,
address,
membership_tier,
updated_at,
case
when is_new_record then true
else false
end as is_active,
updated_at as effective_date,
null as expiry_date  -- will be updated in subsequent runs
from scd_updated
where is_new_record

union all 

select
customer_scd_id,
customer_id,
name,
address,
membership_tier,
updated_at,
false as is_active,
effective_date,
updated_at as expiry_date
from {[this}]
where customer_id in (select customer_id from scd_updates where is_new_record)
and is_active = true;
